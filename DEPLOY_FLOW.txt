DEPLOYMENT STEPS (Run from D:\livingilabs\Sync)
=====================================

--- Step 1: Switch to correct subscription ---
az account set --subscription "58433e7f-268f-4d83-b286-bd805174eec2"

--- Step 2: Login to ACR ---
az acr login --name syncprodacr

--- Step 3: Build images ---
docker build -t syncprodacr.azurecr.io/sync-backend:vX.X.X -t syncprodacr.azurecr.io/sync-backend:latest ./backend
docker build -t syncprodacr.azurecr.io/sync-frontend:vX.X.X -t syncprodacr.azurecr.io/sync-frontend:latest ./frontend

--- Step 4: Push images ---
docker push syncprodacr.azurecr.io/sync-backend:vX.X.X
docker push syncprodacr.azurecr.io/sync-backend:latest
docker push syncprodacr.azurecr.io/sync-frontend:vX.X.X
docker push syncprodacr.azurecr.io/sync-frontend:latest

--- Step 5: Add firewall rule for migrations ---
# Get your public IP first
$myIp = (Invoke-RestMethod -Uri "https://api.ipify.org")
az postgres flexible-server firewall-rule create --resource-group sync-rg --name sync-prod-db --rule-name AllowDeployMachine --start-ip-address $myIp --end-ip-address $myIp

--- Step 6: Run database migrations ---
cd backend
$env:DATABASE_URL = "postgresql://syncadmin:Sync%40Prod2026!@sync-prod-db.postgres.database.azure.com:5432/sync_db?schema=public&sslmode=require"
npx prisma migrate deploy
cd ..

--- Step 7: Update container apps ---
az containerapp update --name sync-prod-backend --resource-group sync-rg --image syncprodacr.azurecr.io/sync-backend:vX.X.X
az containerapp update --name sync-prod-frontend --resource-group sync-rg --image syncprodacr.azurecr.io/sync-frontend:vX.X.X

--- Step 8: Health check (wait ~20 seconds first) ---
# Frontend
Invoke-WebRequest -Uri "https://sync-prod-frontend.delightfulmeadow-6ceb8dcf.centralus.azurecontainerapps.io" -UseBasicParsing -TimeoutSec 15

# Backend logs
az containerapp logs show --name sync-prod-backend --resource-group sync-rg --tail 10

--- Step 9: Clean up firewall rule ---
az postgres flexible-server firewall-rule delete --resource-group sync-rg --name sync-prod-db --rule-name AllowDeployMachine --yes
